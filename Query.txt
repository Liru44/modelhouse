CREATE TABLE qna (
    id BIGSERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    phone VARCHAR(11) NOT NULL,
    content TEXT NOT NULL,
    originate TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE image(
    id BIGSERIAL PRIMARY KEY,
    image_url VARCHAR(500) NOT NULL,
    image_type VARCHAR(50),
    originate TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

SELECT * FROM qna;
